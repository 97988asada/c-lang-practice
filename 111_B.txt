	}
	else if(222 < N && 333 >= N){

	}
	else if(666 < N && 777 >= N){
		printf("777\n");
	}
	else if(777 < N && 888 >= N){

	}
	else if(888 < N && 999 >= N){
		printf("999\n");
	}
	return 0;
}
